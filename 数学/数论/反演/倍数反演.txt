namespace Mobius{
	const int N = 1e6 + 5;
	int mu[N];
	int st[N];
	vector<int> primes;
	void init()
	{
		primes.clear();
		memset(mu, 0, sizeof mu);
		memset(st, 0, sizeof st);
		mu[1] = 1;
		for(int i = 2; i < N; ++ i)
		{
			if(st[i] == 0)
			{
				st[i] = i;
				primes.push_back(i);
				mu[i] = -1;
			}
			for(ll j : primes)
			{
				if(j * i >= N) break;
				st[j * i] = j;
				if(i % j == 0) break;
				else mu[j * i] = - mu[i];
			}
		}
	}

	}
	ll f(ll n, ll UP)
	{
		ll res = 0;
		for(ll d = n; d <= UP; d += n)
		{
			res += mu[d / n] * F(d);
		}
		return res;
	}
}
