namespace Bash{
	/*
	-----#游戏规则#-----
	一个有 n 件物品的物品堆,两个人轮流从这堆物品里取物.
	规定每次至少取走 1 件物品,最多取走 m 件物品，取走最后一件物品的人获胜.
	--------------------
	
	当 n = m + 1 是必败点 :: SG(n) = 0
	当 n = k * (m + 1) 是上一种必败点的堆积,失败 k 次 :: SG(n) = 0
	当 n = k * (m + 1) + s (0 < s < m + 1) 取走 s 个即可转移到SG(n - s) = 0 :: SG(n) > 0
	
	-----#游戏结论#-----
	・ 必败点 : n % (m + 1) == 0
	・ 必胜点 : n % (m + 1) != 0
	*/
	int n, m;
	void init(int _n, int _m)
	{
		n = _n;
		m = _m;
	}
	int	bash()
	{
		return n % (m + 1);
	}
}
