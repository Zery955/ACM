namespace Hash{
	typedef unsigned long long ull;
	const ull base = 131; // 13331;
	const int N = 1e6 + 5;
	ull h[N], p[N];
	inline int id(char c){ return c - 'a';}
	void init(char *s, int n)
	{
		p[0] = 1;
		for(int i = 1; i <= n; ++i)
		{
			int c = id(s[i - 1]);
			h[i] = h[i - 1] * base + c;
			p[i] = p[i - 1] * base;
		}
	}
//******************** ²éÑ¯×Ö´®¹þÏ£Öµ ******************** 
	inline ull query(int l, int r)
	{
		return h[r] - h[l - 1] * p[r - l + 1];
	}
}
